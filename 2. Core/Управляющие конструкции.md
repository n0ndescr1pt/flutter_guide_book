### Stream
```dart
Stream<int> countStream() async* {
  for (int i = 1; i <= 5; i++) {
    yield i;
    await Future.delayed(Duration(seconds: 1));
  }
}

void main() async {
  await for (int value in countStream()) {
    print(value); // Вывод: 1, 2, 3, 4, 5 (с интервалом в 1 секунду)
  }
}
```
